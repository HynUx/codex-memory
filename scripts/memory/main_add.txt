def cmd_search(args):
    """Search existing memories using FTS5 with LIKE fallback."""
    db = init_db()
    limit = args.limit
    offset = args.offset
    kw = args.keywords

    # FTS5 primary path
    rows = db.execute(
        """SELECT e.seq, e.type, e.content, e.topics FROM entries_fts f
           JOIN entries e ON f.rowid = e.seq
           WHERE entries_fts MATCH ? AND e.deleted=0
           ORDER BY rank LIMIT ? OFFSET ?""",
        (kw, limit, offset),
    ).fetchall()

    # LIKE fallback
    if not rows:
        like = f"%{kw}%"
        rows = db.execute(
            "SELECT seq, type, content, topics FROM entries WHERE content LIKE ? AND deleted=0 LIMIT ? OFFSET ?",
            (like, limit, offset),
        ).fetchall()

    if not rows:
        print("未找到相关记忆")
        db.close()
        return 0

    print(f"相关记忆（共 {len(rows)} 条）：")
    for seq, typ, content, topics in rows:
        print(f"  [seq:{seq}] {typ} | {topics} | {content[:60]}...")
    db.close()
    return 0


def cmd_list(args):
    """Browse memories without keyword search."""
    db = init_db()
    limit = args.limit
    rows = db.execute(
        "SELECT seq, type, content, topics FROM entries WHERE deleted=0 ORDER BY seq DESC LIMIT ?",
        (limit,),
    ).fetchall()

    if not rows:
        print("暂无记忆")
        db.close()
        return 0

    print(f"记忆列表（共 {len(rows)} 条）：")
    for seq, typ, content, topics in rows:
        print(f"  [seq:{seq}] {typ} | {topics} | {content[:60]}...")
    db.close()
    return 0
